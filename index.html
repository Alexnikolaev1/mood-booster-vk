import React, { useState, useEffect, useRef, useCallback } from 'react';

 

const MoodBooster = () => {

  const [feeling, setFeeling] = useState('');

  const [isLoading, setIsLoading] = useState(false);

  const [message, setMessage] = useState('');

  const [visualParams, setVisualParams] = useState({

    colors: ['#FFD700', '#FFA500', '#FF69B4', '#00CED1', '#98FB98'],

    speed: 0.6,

    rhythm: 'smooth',

    density: 0.7,

    shape: 'curves',

    energy: 0.8

  });

  const canvasRef = useRef(null);

  const animationRef = useRef(null);

  const timeRef = useRef(0);

  const particlesRef = useRef([]);

 

  // Позитивные сообщения для разных настроений

  const positiveMessages = {

    грустный: "Помни: после дождика всегда выглядывает солнышко! ☀️ Ты сильнее, чем думаешь!",

    усталый: "Отдохни немного, ты заслуживаешь перерыв! 🌸 Завтра будет новый день полный возможностей!",

    злой: "Дыши глубже! 🌊 Эта буря в душе скоро утихнет, и ты снова почувствуешь покой.",

    одинокий: "Ты не один! 💫 Есть люди, которые тебя любят и ценят. Позвони кому-нибудь близкому!",

    тревожный: "Всё будет хорошо! 🦋 Ты справлялся с трудностями раньше, справишься и сейчас!",

    скучный: "Время для приключений! 🎨 Попробуй что-то новое сегодня - нарисуй, спой или просто потанцуй!",

    default: "Ты прекрасен такой, какой есть! ✨ Каждый день - это подарок, и сегодня особенно прекрасный день!"

  };

 

  // Claude API для анализа настроения

  const analyzeFeeling = async (emotion) => {

    if (!emotion.trim()) return;

   

    setIsLoading(true);

   

    const prompt = `Проанализируй эмоцию "${emotion}" и создай позитивное послание для поднятия настроения на русском языке. Также предоставь визуальные параметры как JSON:

{

  "message": "Позитивное послание на русском языке с эмодзи для поднятия настроения",

  "colors": ["#hex1", "#hex2", "#hex3", "#hex4", "#hex5"],

  "speed": число между 0.5 и 1.0,

  "rhythm": "smooth" или "pulsing",

  "density": число между 0.6 и 1.0,

  "shape": "curves" или "circles",

  "energy": число между 0.7 и 1.0

}

 

Используй только яркие, позитивные цвета. Ответь ТОЛЬКО валидным JSON.`;

 

    try {

      const response = await window.claude.complete(prompt);

      const data = JSON.parse(response);

      setVisualParams({

        colors: data.colors || visualParams.colors,

        speed: data.speed || 0.6,

        rhythm: data.rhythm || 'smooth',

        density: data.density || 0.7,

        shape: data.shape || 'curves',

        energy: data.energy || 0.8

      });

      setMessage(data.message || getPositiveMessage(emotion));

    } catch (error) {

      console.error('Ошибка анализа:', error);

      setMessage(getPositiveMessage(emotion));

    } finally {

      setIsLoading(false);

    }

  };

 

  // Получить позитивное сообщение

  const getPositiveMessage = (emotion) => {

    const lowerEmotion = emotion.toLowerCase();

    for (const [key, msg] of Object.entries(positiveMessages)) {

      if (lowerEmotion.includes(key)) {

        return msg;

      }

    }

    return positiveMessages.default;

  };

 

  // Инициализация частиц

  const initializeParticles = useCallback((canvas) => {

    const numParticles = Math.floor(visualParams.density * 150);

    particlesRef.current = Array.from({ length: numParticles }, () => ({

      x: Math.random() * canvas.width,

      y: Math.random() * canvas.height,

      vx: (Math.random() - 0.5) * visualParams.speed * 3,

      vy: (Math.random() - 0.5) * visualParams.speed * 3,

      size: Math.random() * 2 + 1,

      angle: Math.random() * Math.PI * 2,

      angleSpeed: (Math.random() - 0.5) * 0.02,

      life: Math.random(),

      birthTime: Date.now()

    }));

  }, [visualParams]);

 

  // Анимация

  const animate = useCallback((canvas, ctx) => {

    timeRef.current += 0.016 * visualParams.speed;

   

    // Создаём градиентный фон

    const gradient = ctx.createLinearGradient(0, 0, canvas.width, canvas.height);

    gradient.addColorStop(0, 'rgba(255, 105, 180, 0.1)');

    gradient.addColorStop(0.5, 'rgba(0, 206, 209, 0.1)');

    gradient.addColorStop(1, 'rgba(255, 215, 0, 0.1)');

    

    ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';

    ctx.fillRect(0, 0, canvas.width, canvas.height);

   

    ctx.fillStyle = gradient;

    ctx.fillRect(0, 0, canvas.width, canvas.height);

   

    // Анимация частиц

    particlesRef.current.forEach((particle, index) => {

      // Обновление позиции

      if (visualParams.rhythm === 'smooth') {

        particle.x += particle.vx;

        particle.y += particle.vy;

      } else { // pulsing

        const pulse = Math.sin(timeRef.current * 3) * 0.5 + 1;

        particle.x += particle.vx * pulse;

        particle.y += particle.vy * pulse;

      }

     

      // Границы экрана

      if (particle.x < 0) particle.x = canvas.width;

      if (particle.x > canvas.width) particle.x = 0;

      if (particle.y < 0) particle.y = canvas.height;

      if (particle.y > canvas.height) particle.y = 0;

     

      particle.angle += particle.angleSpeed * visualParams.energy;

     

      // Рисование частицы

      ctx.save();

      ctx.translate(particle.x, particle.y);

      ctx.rotate(particle.angle);

     

      const colorIndex = index % visualParams.colors.length;

      const alpha = Math.sin(timeRef.current + index) * 0.3 + 0.7;

      ctx.fillStyle = visualParams.colors[colorIndex] + Math.floor(alpha * 255).toString(16).padStart(2, '0');

      ctx.strokeStyle = visualParams.colors[colorIndex];

      ctx.lineWidth = 2;

     

      if (visualParams.shape === 'curves') {

        // Рисуем красивые кривые

        ctx.beginPath();

        ctx.moveTo(0, 0);

        for (let i = 0; i < 5; i++) {

          const angle = (i / 5) * Math.PI * 2;

          const radius = particle.size * 8 * (Math.sin(timeRef.current + index + i) * 0.3 + 1);

          ctx.lineTo(Math.cos(angle) * radius, Math.sin(angle) * radius);

        }

        ctx.closePath();

        ctx.fill();

        ctx.stroke();

      } else { // circles

        // Рисуем светящиеся круги

        const radius = particle.size * 6 * (Math.sin(timeRef.current * 2 + index) * 0.3 + 1);

        ctx.beginPath();

        ctx.arc(0, 0, radius, 0, Math.PI * 2);

        ctx.fill();

       

        // Внешнее свечение

        ctx.beginPath();

        ctx.arc(0, 0, radius * 1.5, 0, Math.PI * 2);

        ctx.fillStyle = visualParams.colors[colorIndex] + '30';

        ctx.fill();

      }

     

      ctx.restore();

     

      // Обновление жизни частицы

      particle.life += 0.008;

      if (particle.life > 1) {

        particle.life = 0;

        particle.x = Math.random() * canvas.width;

        particle.y = Math.random() * canvas.height;

      }

    });

   

    // Соединения между частицами (создают волшебную сеть)

    for (let i = 0; i < Math.min(particlesRef.current.length, 50); i++) {

      for (let j = i + 1; j < Math.min(particlesRef.current.length, 50); j++) {

        const p1 = particlesRef.current[i];

        const p2 = particlesRef.current[j];

        const dx = p1.x - p2.x;

        const dy = p1.y - p2.y;

        const distance = Math.sqrt(dx * dx + dy * dy);

       

        if (distance < 120) {

          ctx.beginPath();

          ctx.moveTo(p1.x, p1.y);

          ctx.lineTo(p2.x, p2.y);

          const alpha = (1 - distance / 120) * 0.3;

          ctx.strokeStyle = visualParams.colors[0] + Math.floor(alpha * 255).toString(16).padStart(2, '0');

          ctx.lineWidth = 1;

          ctx.stroke();

        }

      }

    }

   

    animationRef.current = requestAnimationFrame(() => animate(canvas, ctx));

  }, [visualParams]);

 

  // Настройка canvas

  useEffect(() => {

    const canvas = canvasRef.current;

    if (!canvas) return;

   

    const ctx = canvas.getContext('2d');

   

    const resizeCanvas = () => {

      canvas.width = window.innerWidth;

      canvas.height = window.innerHeight;

      initializeParticles(canvas);

    };

   

    resizeCanvas();

    window.addEventListener('resize', resizeCanvas);

   

    animate(canvas, ctx);

   

    return () => {

      window.removeEventListener('resize', resizeCanvas);

      if (animationRef.current) {

        cancelAnimationFrame(animationRef.current);

      }

    };

  }, [animate, initializeParticles]);

 

  const handleFeelingSubmit = () => {

    if (feeling.trim()) {

      analyzeFeeling(feeling);

    }

  };

 

  const handleKeyPress = (e) => {

    if (e.key === 'Enter') {

      handleFeelingSubmit();

    }

  };

 

  // Случайный совет

  const getRandomAdvice = () => {

    const advices = [

      "Улыбнись себе в зеркале! 😊",

      "Сделай 10 глубоких вдохов 🌬️",

      "Послушай любимую песню 🎵",

      "Обними кого-то близкого 🤗",

      "Выпей чашку горячего чая ☕",

      "Посмотри на небо и облака ☁️",

      "Вспомни что-то приятное 💭"

    ];

    const randomAdvice = advices[Math.floor(Math.random() * advices.length)];

    setMessage(randomAdvice);

  };

 

  return (

    <div className="w-full h-screen relative overflow-hidden bg-gradient-to-br from-purple-900 via-blue-900 to-indigo-900">

      <canvas

        ref={canvasRef}

        className="absolute inset-0"

      />

     

      {/* Заголовок */}

      <div className="absolute top-8 left-1/2 transform -translate-x-1/2 z-10 text-center">

        <h1 className="text-4xl font-bold text-white mb-2 drop-shadow-lg">

          ✨ Подниматель настроения ✨

        </h1>

        <p className="text-white/80 text-lg">Расскажи, как дела, и получи порцию позитива!</p>

      </div>

     

      {/* Сообщение */}

      {message && (

        <div className="absolute top-1/2 left-1/2 transform -translate-x-1/2 -translate-y-1/2 z-10 max-w-2xl">

          <div className="bg-white/10 backdrop-blur-md rounded-2xl p-8 border border-white/20 shadow-2xl">

            <p className="text-white text-xl text-center leading-relaxed font-medium">

              {message}

            </p>

          </div>

        </div>

      )}

     

      {/* Управление */}

      <div className="absolute bottom-8 left-1/2 transform -translate-x-1/2 z-10">

        <div className="flex flex-col items-center gap-4">

          <div className="flex gap-2">

            <input

              type="text"

              value={feeling}

              onChange={(e) => setFeeling(e.target.value)}

              onKeyPress={handleKeyPress}

              placeholder="Как твоё настроение?"

              className="px-6 py-3 w-80 rounded-full bg-white/10 text-white border border-white/30

                       focus:outline-none focus:border-white/60 backdrop-blur-sm text-center

                       placeholder-white/60 text-lg"

            />

            <button

              onClick={handleFeelingSubmit}

              disabled={isLoading}

              className={`px-8 py-3 rounded-full bg-gradient-to-r from-pink-500 to-purple-500

                       text-white font-semibold transition-all transform hover:scale-105

                       ${isLoading

                         ? 'opacity-50 cursor-not-allowed'

                         : 'hover:shadow-lg hover:from-pink-400 hover:to-purple-400'

                       }`}

            >

              {isLoading ? (

                <span className="flex items-center">

                  <svg className="animate-spin -ml-1 mr-2 h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">

                    <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>

                    <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>

                  </svg>

                  Думаю...

                </span>

              ) : (

                "Поднять настроение! 🚀"

              )}

            </button>

          </div>

         

          <button

            onClick={getRandomAdvice}

            className="px-6 py-2 rounded-full bg-white/10 text-white border border-white/30

                     hover:bg-white/20 transition-colors backdrop-blur-sm"

          >

            💡 Случайный совет

          </button>

        </div>

      </div>

    </div>

  );

};

 

export default MoodBooster;
